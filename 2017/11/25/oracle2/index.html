排序加分页查询

--第一步排序
select * from t_account order by usenum desc ;
-- 第二步生成rownum
select rownum ,t.* from (select * from t_account order by usenum desc) t;
-- 通过rownum进行分页
select * from (select rownum rm ,t.* from (select * from t_account order by usenum desc) t)
where rm>10 and rm <=20;

字符函数

select length('abcd') from dual;-- 求字符串的长度
select substr('adbce',2,2) from dual;-- 截取字符串，第一个数是开始索引，第二个是截取长度
select concat('abc','def') from dual;--拼接字符串
--拼接字符串也可以使用||
select 'abc'||'d'||'ef' from dual;

数值函数

--四舍五入函数round
select round(100.567) from dual;--默认去除小数
select round(100.456,2)from dual;--保留小数后两位

--截取函数trunc
select trunc(100.567) from dual;--去除小数
select trunc(100.567,2) from dual;--截取并保留小数点后两位

--取模mod
select mod(10,5) from dual;--10%5

--日期函数
select sysdate from dual;--使用sysdate系统变量获得当前日期和时间

常用的日期函数

--加月函数add_months
select add_months(sysdate,2) from dual;--在当前日期的基础上加两个月
select add_months(sysdate,-2) from dual;--在当前日期的基础上减两个月
--求所在月最后一天last_day
select last_day(sysdate) from dual;
--使用截取函数截取日期
select trunc(sysdate) from dual;--默认截取后只有年月日
select trunc(sysdate,'yyyy') from dual;--保留年，月日为1
select trunc(sysdate,'mm') from dual;--保留年月，日为1

常用的转换函数

-- 数字转字符串to_char
select to_char(1024) from dual;--将数字1024转为字符串
--日期转字符串to_char
select to_char(sysdate,'yyyy-mm-dd') from dual;--将日期转为指定格式的字符串
--字符串转日期to_date
select to_date('2017-01-01','yyyy-mm-dd') from dual;--将指定格式的日期字符串转为日期
--字符串转数字to_number
select to_number('100') from dual;
select '100'+0 from dual;--将字符串转为数字

空值处理函数

--**空值处理函数nvl 用法：nvl(检测的值，如果为null就用其特换的值）；
select nvl(null,0) from dual;--检测null,为null所以值为0
select nvl(name,'') from users --查询users中的name字段，如果为空设置为‘’

--空值处理函数nvl2 用法：nvl2(检测的值，如果不为null的值，如果为null的值)
select price,minnum,nvl2(maxnum,to_char(maxnum),'不限') 
from t_pricetable 
where ownertypeid=1;

条件取值decode

用法 decode(条件，值1，翻译值，值2，翻译值,...值n,翻译值n,缺省值)
--功能根据条件返回相应值
select 
name,decode(ownertypeid,1,'居民',2,'行政事业单位',3,'商业','其他')
from t_owners
--使用case when then语句实现
select name,(case ownertypeid
             when 1 then '居民'
             when 2 then '行政事业单位'
             when 3 then '商业'
             else '商业'
             end) from t_owners;

行列转换

--需求:按月份统计2012年各个地区的水费
select (select name from t_area where id = areaid) 区域,
       sum(case when month='01' then money else 0 end) 一月,
       sum(case when month='02' then money else 0 end) 二月,
       sum(case when month='03' then money else 0 end) 三月,
       sum(case when month='04' then money else 0 end) 四月,
       sum(case when month='05' then money else 0 end) 五月,
       sum(case when month='06' then money else 0 end) 六月,
       sum(case when month='07' then money else 0 end) 七月,
       sum(case when month='08' then money else 0 end) 八月,
       sum(case when month='09' then money else 0 end) 九月,
       sum(case when month='10' then money else 0 end) 十月,
       sum(case when month='11' then money else 0 end) 十一月,
       sum(case when month='12' then money else 0 end) 十二月
       from t_account where year='2012' group by areaid;

分析函数 可用于排名

--rank 相同的值排名相同,排名跳跃
使用案例: select rank() over(order by usenum desc ),usenum from t_account
--dense_rank 相同的值排名相同,排名连续
使用案例: select dense_rank() over(order by usenum desc ),usenum from t_account
--row_number 返回连续的排名,无论值是否相等 
使用案例: select row_number() over(order by usenum desc ),usenum from t_account

集合运算

集合运算,集合运算就是将两个或者多个结果集组合成为一个结果集
包括:
* union all(并集),返回各个查询的所有记录,包括重复记录
 例:(下面三个使用和它类似)
 select * from t_owners where id<=7
 union all
 select * from t_owners where id>=5

* union(并集),返回各个查询的所有记录,不包括重复记录

* intersect(交集),返回两个查询共有的记录

* minus(差集),返回第一个查询检索出的记录减去第二个查询检索出的记录值后剩余的记录
